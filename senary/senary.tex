% Moon phase name accessor
% #1 = date string YYYY-MM-DD
\newcommand{\moonname}[1]{\csname moon@name@#1\endcsname}

% Moon phase icon (tikz)
% #1 = date string YYYY-MM-DD, #2 = radius (bare number or dimension)
% ponytail: explicit mm units — \pgfmathsetmacro strips units from arg
\newcommand{\moonicon}[2]{%
  \begin{tikzpicture}[baseline=(current bounding box.center)]
    \pgfmathsetmacro{\ph}{\csname moon@phase@#1\endcsname}
    \pgfmathsetmacro{\r}{#2}
    \fill[black!12] (0,0) circle (\r mm);
    \begin{scope}
      \clip (0,0) circle (\r mm);
      \pgfmathsetmacro{\term}{\r - \ph * 4 * \r}
      \fill[moon-yellow] (\term mm, -\r mm) rectangle (\r mm, \r mm);
    \end{scope}
    \draw[ink, thin] (0,0) circle (\r mm);
  \end{tikzpicture}%
}
