% ── Shared size definitions (all values in mm) ──
% Usage: \Size{<spec>}{<prop>}  e.g. \Size{m5}{PW}

\def\Size#1#2{\csname size@#1@#2\endcsname}

% a5s — 110×210mm
\expandafter\def\csname size@a5s@PW\endcsname{110}
\expandafter\def\csname size@a5s@PH\endcsname{210}
\expandafter\def\csname size@a5s@TopGap\endcsname{12}
\expandafter\def\csname size@a5s@RedLine\endcsname{12}
\expandafter\def\csname size@a5s@LastLine\endcsname{39}

% m5 — 67×105mm
\expandafter\def\csname size@m5@PW\endcsname{67}
\expandafter\def\csname size@m5@PH\endcsname{105}
\expandafter\def\csname size@m5@TopGap\endcsname{9}
\expandafter\def\csname size@m5@RedLine\endcsname{7}
\expandafter\def\csname size@m5@LastLine\endcsname{19}
