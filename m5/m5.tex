\documentclass[10pt, twoside]{article}

% ── Page geometry ──
% 67×105mm (m5), no margins
\usepackage[paperwidth=67mm, paperheight=105mm,
            margin=0mm,
            twoside]{geometry}

% ── Packages ──
\usepackage{tikz}

% ── Colors ──
\definecolor{ruleblue}{HTML}{6B9BD2}

% ── Red vertical line at 7mm from left ──
\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture, overlay]
    \draw[red!60, line width=0.4pt]
      ([xshift=7mm] current page.south west) --
      ([xshift=7mm] current page.north west);
  \end{tikzpicture}%
}

% ── Page style ──
\pagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\parskip}{0pt}
\setlength{\topskip}{0pt}

% ── Ruled page macro ──
% Blue horizontal lines at 6mm spacing, 9mm top gap, 17 lines
\newcommand{\ruledpage}{%
  \begin{tikzpicture}[remember picture, overlay]
    \foreach \n in {0, 1, ..., 16} {
      \pgfmathsetmacro{\y}{-9 - \n * 6}
      \draw[ruleblue!45, line width=0.3pt]
        ([yshift=\y mm] current page.north west) --
        ([yshift=\y mm] current page.north east);
    }
  \end{tikzpicture}%
  \null
}
