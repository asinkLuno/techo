% ──────────────────────────────────────────
% Night Owl cozyca — triangular hourglass, numbers 0–26 colored by night sky
% Build: uv run python nightowl.py --size cozyca && cd
% night-owl-cozyca && xelatex night-owl-cozyca.tex
% ──────────────────────────────────────────
\documentclass[10pt, twoside]{article}

% ── Size ──
\input{../sizes.tex}

% ── Page geometry ──
\usepackage[paperwidth=\Size{cozyca}{PW} mm, paperheight=\Size{cozyca}{PH} mm,
  margin=0mm,
twoside]{geometry}

% ── Packages ──
\usepackage{tikz}
\usepackage{fontspec}
\usepackage{xcolor}

% ── Colors ──
\definecolor{fluor}{HTML}{39FF14}
\definecolor{cross}{HTML}{FF0000}

% ── Background: dot-grid ──
\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture, overlay]
    % Dot grid
    \foreach \x in {0.5,3.5,...,99.5} {
      \foreach \y in {-1.5,-4.5,...,-89.5} {
        \fill[fluor!50] (\x mm, \y mm) circle (0.35mm);
      }
    }
    % Center dot → red
    \fill[cross!50] (50.0mm, -45.0mm) circle (0.35mm);
  \end{tikzpicture}%
}

% ── 3270 Nerd Font (retro terminal numerals) ──
\newfontfamily{\nightowlfont}{3270 Nerd Font Mono}[
  Scale=1.0,
]

% ── Page style ──
\pagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\parskip}{0pt}

% ──────────────────────────────────────────
\begin{document}
\input{content.tex}
\end{document}
