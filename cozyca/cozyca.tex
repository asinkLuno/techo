\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture, overlay]

    % ====== Hardcoded positions (mm from page top-left) ======
    % Table 1: top=-8, header=-13, bottom=-38
    % Table 2: top=-43, header=-48, bottom=-73
    % Label divider at x=15, grid right at x=95

    % ====== Table 1 (days 1–15) ======
    \draw[black!45, line width=0.3pt]
    ([xshift=5mm, yshift=-8mm] current page.north west) rectangle
    ([xshift=95mm, yshift=-38mm] current page.north west);
    \draw[black!40, line width=0.25pt]
    ([xshift=15mm, yshift=-8mm] current page.north west) --
    ([xshift=15mm, yshift=-38mm] current page.north west);
    \draw[black!40, line width=0.25pt]
    ([xshift=5mm, yshift=-13mm] current page.north west) --
    ([xshift=95mm, yshift=-13mm] current page.north west);
    \foreach \c in {0,...,14} {
      \pgfmathsetmacro{\x}{15 + (\c+0.5)*5}
      \node[font=\fontsize{4.5}{5.5}\selectfont, text=black!55]
      at ([xshift=\x mm, yshift=-10.5mm] current page.north west)
      {\the\numexpr1+\c\relax};
    }
    \foreach \r in {0,...,5} {
      \pgfmathsetmacro{\y}{-13 - \r*5}
      \draw[black!25, line width=0.12pt]
      ([xshift=5mm, yshift=\y mm] current page.north west) --
      ([xshift=95mm, yshift=\y mm] current page.north west);
    }
    \foreach \c in {0,...,16} {
      \pgfmathsetmacro{\x}{15 + \c*5}
      \draw[black!25, line width=0.12pt]
      ([xshift=\x mm, yshift=-8mm] current page.north west) --
      ([xshift=\x mm, yshift=-38mm] current page.north west);
    }

    % ====== Table 2 (days 16–31) ======
    \draw[black!45, line width=0.3pt]
    ([xshift=5mm, yshift=-43mm] current page.north west) rectangle
    ([xshift=95mm, yshift=-73mm] current page.north west);
    \draw[black!40, line width=0.25pt]
    ([xshift=15mm, yshift=-43mm] current page.north west) --
    ([xshift=15mm, yshift=-73mm] current page.north west);
    \draw[black!40, line width=0.25pt]
    ([xshift=5mm, yshift=-48mm] current page.north west) --
    ([xshift=95mm, yshift=-48mm] current page.north west);
    \foreach \c in {0,...,15} {
      \pgfmathsetmacro{\x}{15 + (\c+0.5)*5}
      \node[font=\fontsize{4.5}{5.5}\selectfont, text=black!55]
      at ([xshift=\x mm, yshift=-45.5mm] current page.north west)
      {\the\numexpr16+\c\relax};
    }
    \foreach \r in {0,...,5} {
      \pgfmathsetmacro{\y}{-48 - \r*5}
      \draw[black!25, line width=0.12pt]
      ([xshift=5mm, yshift=\y mm] current page.north west) --
      ([xshift=95mm, yshift=\y mm] current page.north west);
    }
    \foreach \c in {0,...,16} {
      \pgfmathsetmacro{\x}{15 + \c*5}
      \draw[black!25, line width=0.12pt]
      ([xshift=\x mm, yshift=-43mm] current page.north west) --
      ([xshift=\x mm, yshift=-73mm] current page.north west);
    }

  \end{tikzpicture}%
}
