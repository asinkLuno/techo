% Shared dot-grid background — green dots, red center
% Margins: set \DTOP/\DBOT/\DLEFT/\DRIGHT before \input to override 0 defaults
\providecommand{\DTOP}{0}
\providecommand{\DBOT}{0}
\providecommand{\DLEFT}{0}
\providecommand{\DRIGHT}{0}

\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture, overlay]
    % Center dot position first — grid snaps to this
    \pgfmathsetmacro{\cx}{\PW/2}
    \pgfmathsetmacro{\cy}{-\PH/2}
    \pgfmathsetmacro{\gs}{3}
    % Grid offset ranges from center (integer steps of \gs)
    \pgfmathsetmacro{\nmin}{ceil((\DLEFT+0.5-\cx)/\gs)}
    \pgfmathsetmacro{\nmax}{floor((\PW-\DRIGHT-0.5-\cx)/\gs)}
    \pgfmathsetmacro{\mmin}{ceil((-\PH+\DBOT+1.5-\cy)/\gs)}
    \pgfmathsetmacro{\mmax}{floor((-\DTOP-1.5-\cy)/\gs)}
    % Green dots (skip center — red dot goes there)
    \foreach \n in {\nmin,...,\nmax} {
      \foreach \m in {\mmin,...,\mmax} {
        \pgfmathsetmacro{\dx}{\cx + \n*\gs}
        \pgfmathsetmacro{\dy}{\cy + \m*\gs}
        \pgfmathsetmacro{\iscenter}{\n==0 && \m==0 ? 1 : 0}
        \ifnum\iscenter=0
        \fill[NeonGreen!50] (\dx mm, \dy mm) circle (0.35mm);
        \fi
      }
    }
    % Center dot → red
    \fill[IronOxideRed!50] (\cx mm, \cy mm) circle (0.35mm);
  \end{tikzpicture}%
}
