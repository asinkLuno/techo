% Xianzhang — French ruled (Séyès) notebook (shared template, all sizes)
% Size is set by \EDITION in each edition's generated wrapper.
% Build: uv run python xianzhang.py xianzhang-<size>
%        cd xianzhang-<size> && xelatex xianzhang-<size>.tex
% (run from the edition dir; sizes.tex is at the repo root)
% ──────────────────────────────────────────
\documentclass[10pt, twoside]{article}

% ── Size ──
\input{../../src/sizes.tex}

% ── Page geometry ──
\usepackage[paperwidth=\Size{\EDITION}{PW} mm,
  paperheight=\Size{\EDITION}{PH} mm,
  margin=0mm,
twoside]{geometry}

% ── Packages ──
\usepackage{tikz}
\input{../../src/colors.tex}

% ── Red vertical margin line (odd→right, even→left) ──
\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture, overlay]
    \ifodd\value{page}
    \draw[IronOxideRed!60, line width=0.4pt]
    ([xshift=-\Size{\EDITION}{RedLine} mm] current page.south east) --
    ([xshift=-\Size{\EDITION}{RedLine} mm] current page.north east);
    \else
    \draw[IronOxideRed!60, line width=0.4pt]
    ([xshift=\Size{\EDITION}{RedLine} mm] current page.south west) --
    ([xshift=\Size{\EDITION}{RedLine} mm] current page.north west);
    \fi
  \end{tikzpicture}%
}

% ── Page style ──
\pagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\parskip}{0pt}
\setlength{\topskip}{0pt}

% ──────────────────────────────────────────
\begin{document}
\input{content.tex}
\end{document}
