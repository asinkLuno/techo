% the photo field: a dashed 2:3 outline labelled STAMP HERE.
% Frame is slightly larger than the poster so the dashed outline is visible.
\newcommand{\posterfield}{%
  \node[draw=Outline, dashed, line width=0.5pt, minimum width=\PosterW mm,
    minimum height=\PosterH mm,
  inner sep=0pt] (frame)
  {\color{Outline}\fontsize{\StampPt pt}{\dimexpr\StampPt
  pt*6/5\relax}\selectfont\bfseries STAMP HERE};%
}
% placeholder (no poster): labelled dashed field, ref code pinned under it
\newcommand{\posterbox}{
  \begin{tikzpicture}
    \posterfield
    \node[below=2mm of frame] {\caplabel{[ REF\_IMG\_001 ]}};
\end{tikzpicture}}
% fetched poster: the print is pasted inside a slightly larger dashed frame,
% nudged and tilted subtly so it reads as hand-pasted rather than misaligned.
\newcommand{\posterimage}[1]{
  \begin{tikzpicture}
    \posterfield
    \pgfmathsetmacro{\pr}{random()*0.8}
    \node[inner sep=0pt, draw=Outline, line width=0.3pt, rotate=\pr] (photo)
    at ([shift={(0.6mm,-0.4mm)}]frame.center)
    {\includegraphics[width=\PosterW mm,height=\PosterH
    mm,keepaspectratio]{#1}};
    \node[below=2mm of photo] {\caplabel{[ REF\_IMG\_001 ]}};
\end{tikzpicture}}

% rotated red stamp (used for the DATE field) — slight random
% shift and rotation jitter so it reads as hand-stamped.
\newcommand{\datestamp}[1]{\tikz[baseline=-1.2ex]{
    \pgfmathsetmacro{\jx}{random()*1.2}
    \pgfmathsetmacro{\jy}{random()*1.2}
    \pgfmathsetmacro{\jr}{-3+random()*0.8}
    \node[draw=StampRed, line width=0.8pt,
      rounded corners=1pt, inner sep=3pt, rotate=\jr,
      xshift=\jx mm, yshift=\jy mm,
      font={\fontsize{\StampPt pt}{\dimexpr\StampPt
pt*6/5\relax}\selectfont\bfseries\color{StampRed}}]{#1};}}

% Dashed circle check-box — baseline at circle centre so \raisebox can
% vertically centre it between the label and the date blank.
\newcommand{\epbox}{\tikz[baseline=0pt]{\node[draw=Outline, dashed,
      line width=0.4pt,
    fill=WhiteFill, circle, minimum size=\CircleSize mm, inner sep=0pt]
at (0,0) {};}}
% Write-in date blank — ruled lines sit below the baseline so handwriting
% rests on them. The \fontsize keeps slashes and spacing consistent with the
% label above; \rule width is balanced so 5–6 stamps fit on an A5 row.
\newcommand{\dateblank}{{\color{Outline}\fontsize{\LabelPt
    pt}{\dimexpr\LabelPt pt*6/5\relax}\selectfont
    \rule[-0.5ex]{\DateblankSeg mm}{0.4pt}\hspace{1pt}/\hspace{1pt}%
    \rule[-0.5ex]{\DateblankSeg mm}{0.4pt}\hspace{1pt}/\hspace{1pt}%
\rule[-0.5ex]{\DateblankSeg mm}{0.4pt}}}
% a check-box, its label, and a __/__/__ to fill the viewing date
% \raisebox centres the circle vertically between the two text lines.
\newcommand{\epitem}[1]{%
  \raisebox{\Raisebox pt}{\epbox}\enspace
  \begin{tabular}[b]{@{}l@{}}%
    {\fontsize{\LabelPt pt}{\dimexpr\LabelPt pt*6/5\relax}\selectfont
    EP#1}\\[1pt]%
    \dateblank%
  \end{tabular}%
}
% a check-box labelled SEEN, with a __/__/__ viewing-date field
\newcommand{\seenitem}{%
  \raisebox{\Raisebox pt}{\epbox}\enspace
  \begin{tabular}[b]{@{}l@{}}%
    {\fontsize{\LabelPt pt}{\dimexpr\LabelPt pt*6/5\relax}\selectfont
    SEEN}\\[1pt]%
    \dateblank%
  \end{tabular}%
}

% internal field notes — a memo area that fills the remaining page height/width,
% ruled with the same 3 mm green-dot grid as green_dot.  Height = leftover space
% on the page minus what the footer (ref code + FILE_ACTIVE) needs.
\newcommand{\notesbox}{%
  \par
  \noindent\caplabel{INTERNAL FIELD NOTES:}\vspace{4pt}\par
  \setlength{\notesheight}{\dimexpr\textheight-\pagetotal-44pt\relax}%
  \ifdim\notesheight<60pt\relax\setlength{\notesheight}{60pt}\relax\fi
  \noindent
  \begin{tikzpicture}
    \useasboundingbox (0,-\notesheight) rectangle (\linewidth, 0);
    \fill[white] (0,-\notesheight) rectangle (\linewidth, 0);
    % Green-dot grid (3 mm spacing, same algorithm as dot-grid.tex):
    % dots radiate from the box centre so left/right and top/bottom
    % margins are guaranteed equal.
    \pgfmathsetmacro{\bw}{\linewidth/1mm}         % box width (mm)
    \pgfmathsetmacro{\bh}{\notesheight/1mm}       % box height (mm)
    \pgfmathsetmacro{\cx}{\bw/2}                  % centre x
    \pgfmathsetmacro{\cy}{-\bh/2}                 % centre y
    \pgfmathsetmacro{\gs}{3}                      % grid spacing (mm)
    \pgfmathsetmacro{\nmin}{ceil((1.5-\cx)/\gs)}
    \pgfmathsetmacro{\nmax}{floor((\bw-1.5-\cx)/\gs)}
    \pgfmathsetmacro{\mmin}{ceil((-\bh+1.5-\cy)/\gs)}
    \pgfmathsetmacro{\mmax}{floor((-1.5-\cy)/\gs)}
    \foreach \n in {\nmin,...,\nmax} {
      \foreach \m in {\mmin,...,\mmax} {
        \pgfmathsetmacro{\dx}{\cx+\n*\gs}
        \pgfmathsetmacro{\dy}{\cy+\m*\gs}
        \pgfmathsetmacro{\iscenter}{\n==0 && \m==0 ? 1 : 0}
        \ifnum\iscenter=0
        \fill[NeonGreen!50] (\dx mm, \dy mm) circle (0.35mm);
        \fi
      }
    }
    % Red centre dot
    \fill[IronOxideRed!50] (\cx mm, \cy mm) circle (0.35mm);
    \draw[Outline, line width=0.4pt] (0,-\notesheight) rectangle
    (\linewidth, 0);
\end{tikzpicture}\par}

% rotated FILE_ACTIVE status stamp — slight random drift for a
% hand-stamped look, same as \datestamp.
\newcommand{\filestamp}[1]{\tikz[baseline=-1ex]{
    \pgfmathsetmacro{\jx}{random()*1.5}
    \pgfmathsetmacro{\jy}{random()*1.5}
    \pgfmathsetmacro{\jr}{-4+random()*0.8}
    \node[draw=StampRed, line width=1.4pt,
      inner sep=4pt, rotate=\jr,
      xshift=\jx mm, yshift=\jy mm,
      font={\capfont\fontsize{\dimexpr\StampPt
        pt*6/5\relax}{\dimexpr\StampPt
pt*3/2\relax}\selectfont\bfseries\color{StampRed}}]{#1};}}

% aesthetic footer: ref code / classification (left) + status stamp (right)
\newcommand{\reportfooter}[1]{%
  \vspace{10pt}\noindent
  \begin{minipage}[c]{0.6\linewidth}\raggedright
    {\fontsize{\FooterPt pt}{\dimexpr\FooterPt pt*6/5\relax}\selectfont\itshape
    Ref Code: #1\\Classification: Top Secret / Eyes Only}
  \end{minipage}\hfill
  \begin{minipage}[c]{0.35\linewidth}\raggedleft
    \filestamp{FILE\_ACTIVE}
\end{minipage}\par}
