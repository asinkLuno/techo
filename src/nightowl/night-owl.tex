% ──────────────────────────────────────────
% Night Owl — triangular hourglass, numbers 0–26 colored by night sky
% Size is set by \EDITION in each edition's generated wrapper.
% Build: uv run python nightowl.py --size <m5|cozyca>
%        cd night-owl-<size> && xelatex night-owl-<size>.tex
% (run from the edition dir; sizes.tex is at the repo root)
% ──────────────────────────────────────────
\documentclass[10pt, twoside]{article}

% ── Size ──
\input{../../src/sizes.tex}
% Materialize current size dims as plain numbers (dot grid derives from these)
\edef\PW{\Size{\EDITION}{PW}}
\edef\PH{\Size{\EDITION}{PH}}

% ── Page geometry ──
\usepackage[paperwidth=\PW mm, paperheight=\PH mm,
  margin=0mm,
twoside]{geometry}

% ── Packages ──
\usepackage{tikz}
\input{../../src/colors.tex}
\usepackage{fontspec}
\usepackage{xcolor}

% ── Background: dot-grid (geometry derived from page size) ──
\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture, overlay]
    \pgfmathsetmacro{\gxmax}{\PW-0.5}
    \pgfmathsetmacro{\gymin}{-\PH+1.5}
    \pgfmathsetmacro{\cx}{\PW/2}
    \pgfmathsetmacro{\cy}{-\PH/2}
    % Dot grid
    \foreach \x in {0.5,3.5,...,\gxmax} {
      \foreach \y in {-1.5,-4.5,...,\gymin} {
        \fill[NeonGreen!50] (\x mm, \y mm) circle (0.35mm);
      }
    }
    % Center dot → red
    \fill[IronOxideRed!50] (\cx mm, \cy mm) circle (0.35mm);
  \end{tikzpicture}%
}

% ── 3270 Nerd Font (retro terminal numerals) ──
\newfontfamily{\nightowlfont}{3270 Nerd Font Mono}[
  Scale=1.0,
]

% ── Page style ──
\pagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\parskip}{0pt}

% ──────────────────────────────────────────
\begin{document}
\input{content.tex}
\end{document}
